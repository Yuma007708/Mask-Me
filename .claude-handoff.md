### B. 完走したらローカル集計（GitHub Actions の Aggregate でも summary に出る）

```bash
SCRATCH=/tmp/ci-results
mkdir -p "$SCRATCH"
gh run download 28450221341 --repo Yuma007708/Mask-Me --dir "$SCRATCH"
python3 <<'PY'
import json, glob, os
root = "/tmp/ci-results"
rows = []
for f in sorted(glob.glob(os.path.join(root, "result-*/result.json"))):
    text = open(f).read()
    s, e = text.find("{"), text.find("}", text.find("{"))
    if s < 0 or e < 0: continue
    rows.append(json.loads(text[s:e+1]))
rows.sort(key=lambda d:(d["video"], d["backend"]))
print(f"{'video':<5} {'backend':<13} {'total':>5} {'hit':>5} {'rate':>6} {'lowCy':>6} {'lowRate':>7} {'baseline':>8}  delta")
for d in rows:
    delta = (d["rate"]-d["baseline"])*100
    print(f'{d["video"]:<5} {d["backend"]:<13} {d["total"]:>5} {d["hit"]:>5} {d["rate"]*100:>5.1f}% {d["lowCy"]:>6} {d["lowRate"]*100:>6.1f}% {d["baseline"]*100:>7.0f}%  {delta:+.1f}pp')
PY
```
